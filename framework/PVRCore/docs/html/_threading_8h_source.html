356  // This function will test if there is spend an unblock token. Should be called once whenever a dequeue
357  inline bool tryUnblockAndReturnTrueIfUnblocked()
358  {
359  // times out
360  if (_isUnblocking)
361  {
362  // Spend an "unblock token"
363  int32_t howManyUnblocks = _unblockingCounter--;
364  if (howManyUnblocks == 1)
365  {
366  // Finish unblocking if all tokens are spent
367  _isUnblocking = false;
368  } // Last one!
369  if (howManyUnblocks > 0)
370  {
371  // If you didn't spend a token that was not there, break free, you are unblocked
372  return true;
373  }
374  ++_unblockingCounter; // Nope, not unblocking, at least not for THIS thread. Revert your change, and keep waiting - token is already spent.
375  }
376  return false;
377  }
