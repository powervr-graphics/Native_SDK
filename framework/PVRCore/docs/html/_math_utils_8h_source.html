274 inline uint64_t generatePascalTriangleRow(const uint32_t row, std::vector<uint64_t>& pascalCoefficients)
275 {
276  // Each entry of any given row of the Pascal Triangle is constructed by adding the number above and to the left with the number above and to the right.
277  // Entries which fall outside of the Pascal Triangle are treated as having a 0 value. The first row consists of a single entry with a value of 1.
278  // The following shows the first 4 rows of the Pascal Triangle
279  // 1 row 0 ... sum = 1
280  // 1 1 row 1 ... sum = 2
281  // 1 2 1 row 2 ... sum = 4
282  // 1 3 3 1 row 3 ... sum = 8
283  pascalCoefficients.emplace_back(1);
284  uint64_t sum = pascalCoefficients.back();
285  for (uint32_t i = 0; i < row; i++)
286  {
287  uint64_t val = pascalCoefficients[i] * (row - i) / (i + 1);
288  pascalCoefficients.emplace_back(val);
289  sum += pascalCoefficients.back();
290  }
291 
292  return sum;
293 }

300 inline void adjustOffsetsAndWeightsForLinearSampling(const uint32_t halfCoefficientsMinusOne, std::vector<double>& weights, std::vector<double>& offsets)
301 {
302  std::vector<double> adjustedWeights;
303  std::vector<double> adjustedOffsets;
304 
305  // if kernel size minus 1 is divisible by 2 then we have a central sample with offset 0
306  if (halfCoefficientsMinusOne % 2 == 0)
307  {
308  uint32_t i = 0;
309  for (; i < static_cast<uint32_t>(std::max(static_cast<int32_t>(halfCoefficientsMinusOne) - 1, 0)); i += 2)
310  {
311  adjustedWeights.emplace_back(weights[i] + weights[i + 1]);
312  double adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
313  adjustedOffsets.emplace_back(adjustedOffset);
314  }
315 
316  adjustedWeights.emplace_back(weights[halfCoefficientsMinusOne]);
317  adjustedOffsets.emplace_back(0.0f);
318 
319  for (i = halfCoefficientsMinusOne + 1; i < offsets.size(); i += 2)
320  {
321  adjustedWeights.emplace_back(weights[i] + weights[i + 1]);
322  double adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
323  adjustedOffsets.emplace_back(adjustedOffset);
324  }
325  }
326  else // otherwise we have to duplicate the central sample *but* this means we can handle 3x3 using 2x2 samples
327  {
328  uint32_t i = 0;
329  for (; i < halfCoefficientsMinusOne; i += 2)
330  {
331  double adjustedOffset = 0.0;
332  if (i == halfCoefficientsMinusOne - 1)
333  {
334  adjustedWeights.emplace_back(weights[i] + weights[i + 1] * 0.5);
335  adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1] * 0.5)) / adjustedWeights.back();
336  }
337  else
338  {
339  adjustedWeights.emplace_back(weights[i] + weights[i + 1]);
340  adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
341  }
342  adjustedOffsets.emplace_back(adjustedOffset);
343  }
344 
345  for (i = halfCoefficientsMinusOne; i < offsets.size(); i += 2)
346  {
347  double adjustedOffset = 0.0;
348  if (i == halfCoefficientsMinusOne)
349  {
350  adjustedWeights.emplace_back(weights[i] * 0.5 + weights[i + 1]);
351  adjustedOffset = ((offsets[i] * weights[i] * 0.5) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
352  }
353  else
354  {
355  adjustedWeights.emplace_back(weights[i] + weights[i + 1]);
356  adjustedOffset = ((offsets[i] * weights[i]) + (offsets[i + 1] * weights[i + 1])) / adjustedWeights.back();
357  }
358  adjustedOffsets.emplace_back(adjustedOffset);
359  }
360  }
361 
362  offsets.clear();
363  weights.clear();
364 
365  for (uint32_t i = 0; i < adjustedWeights.size(); i++)
366  {
367  offsets.emplace_back(adjustedOffsets[i]);
368  weights.emplace_back(adjustedWeights[i]);
369  }

390 inline void generateGaussianKernelWeightsAndOffsets(uint32_t kernelSize, bool truncateCoefficients, bool useLinearSamplerOptimization, std::vector<double>& weights,
391  std::vector<double>& offsets, float minimumAcceptableCoefficient = 0.0001f)
392 {
393  // Odd kernel sizes are a requirement
394  assert((kernelSize % 2) == 1);
395 
396  // The starting row of the Pascal Triangle being used
397  uint32_t pascalRow = kernelSize - 1;
398 
399  // The number of coefficients minus 1 halved
400  // This variable is used to get the index of the coefficient used for an offset of 0
401  // (numCoefficients - 1) / 2
402  uint32_t halfCoefficientsMinusOne = pascalRow / 2;
403 
404  // stores the set of pascal coefficients
405  std::vector<uint64_t> pascalCoefficients;
406 
407  // the number of coefficients skipped due to minimum coefficient checks
408  uint32_t numCoefficientsSkipped = 0;
409 
410  // The simple case
411  if (!truncateCoefficients)
412  {
413  // Generate the Pascal Triangle row and calculate the sum of the row
414  uint64_t pascalSum = generatePascalTriangleRow(pascalRow, pascalCoefficients);
415  // Store the Pascal Triangle coefficients for the given row
416  addCoefficients(pascalSum, halfCoefficientsMinusOne, static_cast<uint32_t>(pascalCoefficients.size()), pascalCoefficients, weights, offsets);
417  }
418  else
419  // Ignoring negligible coefficients - we'll now attempt to find a row which provides enough coefficients for the given kernel size whilst not falling below the
420  // given minimal coefficient value.
421  {
422  uint32_t currentRow = pascalRow;
423  bool foundSuitableRow = false;
424 
425  // only accept rows where we have kernelSize coefficients larger than the min coefficient
426  while (!foundSuitableRow)
427  {
428  // clear the offsets and weights
429  offsets.clear();
430  weights.clear();
431 
432  // clear the set of pascal triangle coefficients
433  pascalCoefficients.clear();
434 
435  // get the pascal coefficients for the current row of the triangle
436  uint64_t pascalSum = generatePascalTriangleRow(currentRow, pascalCoefficients);
437 
438  halfCoefficientsMinusOne = currentRow / 2;
439 
440  // check how many of the coefficients are negligible in size and therefore should be ignored
441  numCoefficientsSkipped = 0;
442  for (uint32_t i = halfCoefficientsMinusOne; i < pascalCoefficients.size(); i++)
443  {
444  double currentWeight = static_cast<double>(pascalCoefficients[i]) / pascalSum;
445  if (currentWeight < minimumAcceptableCoefficient)
446  {
447  numCoefficientsSkipped++;
448  }
449  }
450 
451  // if there aren't enough coefficients left to fulfill the requirements for the kernel size then continue to the next row
452  if ((halfCoefficientsMinusOne + 1) - numCoefficientsSkipped < (pascalRow / 2 + 1))
453  {
454  currentRow += 2;
455  foundSuitableRow = false;
456  continue;
457  }
458 
459  // if negligible coefficients are to be removed we must also update the overall sums used to give them weighting
460  // otherwise repeated blurring will result in darkening of the image
461  uint64_t adjustedPascalSum = pascalSum;
462  // the extra coefficients which aren't needed which do match the conditions for non-negligibility but would result in us taking extra coefficients
463  uint64_t unrequiredCoefficients = ((pascalCoefficients.size() - kernelSize - (numCoefficientsSkipped * 2)) / 2);
464  for (uint32_t i = 0; i < numCoefficientsSkipped + unrequiredCoefficients; i++)
465  {
466  adjustedPascalSum -= 2 * pascalCoefficients[pascalCoefficients.size() - 1 - i];
467  }
468 
469  // add the non-negligible coefficients to the weights and offsets buffers
470  uint32_t numCoefficients = static_cast<uint32_t>(pascalCoefficients.size() - (2 * (numCoefficientsSkipped + unrequiredCoefficients)));
471  addCoefficients(adjustedPascalSum, halfCoefficientsMinusOne, numCoefficients, pascalCoefficients, weights, offsets);
472  halfCoefficientsMinusOne = static_cast<uint32_t>((offsets.size() - 1) / 2);
473  foundSuitableRow = true;
474  }
475  }
476 
477  // If using the Linear Sampling optimisation then adjust the Gaussian weights and offsets accordingly
478  if (useLinearSamplerOptimization)
479  {
480  adjustOffsetsAndWeightsForLinearSampling(halfCoefficientsMinusOne, weights, offsets);
481  }
482 }

Generates a set of Gaussian weights and offsets based on the given configuration values. This function makes use of the Pascal Triangle for calculating the Gaussian distribution. The Gaussian function is a distribution function of the normal distribution who's discrete equivalent is the binomial distribution for which the Pascal Triangle models. The Pascal Triangle provides us with a convenient and efficient mechanism for calculating the Gaussian weights and offsets required. Our method of generating Gaussian weights and offsets was inspired by http://rastergrid.com/blog/2010/09/efficient-gaussian-blur-with-linear-sampling/ .
