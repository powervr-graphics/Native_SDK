45 struct IRefCountEntry
46 {
47  template<typename>
48  friend class RefCountedResource;
49  template<typename>
50  friend class EmbeddedRefCountedResource;
51  template<typename>
52  friend class RefCountedWeakReference;

54  friend class EmbeddedRefCount;
55 
56 private:
57  mutable volatile std::atomic<int32_t> count_;
58  mutable volatile std::atomic<int32_t> weakcount_;
59  std::mutex mylock;
60 
61 public:
62  int32_t count() { return count_; }
63  int32_t weakcount() { return weakcount_; }
64 
65 protected:
67  void increment_count()
68  {
69  if (std::atomic_fetch_add(&count_, 1) == 0)
70  {
71  std::lock_guard<std::mutex> lockguard(mylock);
72  if (count_.load() == 0) { throw std::runtime_error("RefCounted::increment_count: Tried to increment the count of an object but it had already been destroyed!"); }
73  }
74  }
76  void increment_weakcount()
77  {
78  if (std::atomic_fetch_add(&weakcount_, 1) == 0)
79  {
80  std::lock_guard<std::mutex> lockguard(mylock);
81  if (weakcount_.load() == 0) { throw std::runtime_error("RefCounted::increment_count: Tried to increment the count of an object but it had already been destroyed!"); }
82  }
83  }
84 
87  void decrement_count()
88  {
89  bool deleter = false; // Track, if it was THIS object that was the one to do the deletion!
90  int32_t cnt = 0;
91  {
92  if ((cnt = std::atomic_fetch_add(&count_, -1)) == 1)
93  {
94  std::lock_guard<std::mutex> lockguard(mylock);
95  destroyObject();
96  if (weakcount() == 0) { deleter = true; }
97  }
98  }
99  if (deleter) { deleteEntry(); }
100  }
101 
104  void decrement_weakcount()
105  {
106  bool deleter = false; // DO NOT DESTROY THE OBJECT WHILE THE MUTEX IS USED!
107  if (((std::atomic_fetch_add(&weakcount_, -1)) == 1))
108  {
109  std::lock_guard<std::mutex> lockguard(mylock);
110  if (count_.load() == 0) { deleter = true; }
111  }
112  if (deleter) { deleteEntry(); }
113  }
114 
115  virtual void deleteEntry() = 0;
116  virtual void destroyObject() = 0;
117  virtual ~IRefCountEntry() {}
118  IRefCountEntry() : count_(1), weakcount_(0) {}
