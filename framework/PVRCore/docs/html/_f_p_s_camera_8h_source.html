137  void updateRUL() const
138  {
139  if (_isOrientationDirty)
140  {
141  // If the pitch and yaw angles are in degrees,
142  // they need to be converted to radians. Here
143  // I assume the values are already converted to radians.
144  const float cosPitch = cos(glm::radians(_pitch));
145  const float sinPitch = sin(glm::radians(_pitch));
146  const float cosYaw = cos(glm::radians(_yaw));
147  const float sinYaw = sin(glm::radians(_yaw));
148 
149  _right = { cosYaw, 0, -sinYaw };
150  _up = { sinYaw * sinPitch, cosPitch, cosYaw * sinPitch };
151  _look = { sinYaw * cosPitch, -sinPitch, cosPitch * cosYaw };
152  _isOrientationDirty = false;
