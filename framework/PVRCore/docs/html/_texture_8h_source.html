56 inline float growFloat(const uint32_t& value, const uint8_t& mantissaBits, const uint8_t& exponentBits, const uint8_t& exponentBias, const bool hasSign /*=true*/)
57 {
58  // Union to avoid messy casting.
59  union
60  {
61  float fVal;
62  uint32_t uiVal;
63  } intToFloat;
64 
65  // Float 32 constants.
66  static const uint8_t f32expBias = 127;
67  static const uint8_t f32signBits = 1;
68  static const uint8_t f32expBits = 8;
69  static const uint8_t f32manBits = 23;
70  static const uint8_t f32totBits = 32;
71 
72  // Get the # of sign bits (always 1 or 0)
73  uint8_t signBits = static_cast<uint8_t>(hasSign ? 1 : 0);
74 
75  // Work out the total bits
76  uint8_t totalBits = static_cast<uint8_t>(mantissaBits + exponentBits + signBits);
77 
78  // Generate the masks for each bit.
79  uint32_t signMask = static_cast<uint8_t>(((1 << signBits) - 1) << (exponentBits + mantissaBits));
80  uint32_t expoMask = static_cast<uint8_t>(((1 << exponentBits) - 1) << mantissaBits);
81  uint32_t mantMask = static_cast<uint8_t>(((1 << mantissaBits) - 1));
82 
83  // Get original exponent.
84  uint32_t originalExponent = ((value & expoMask) >> (mantissaBits));
85  bool originalIsNaN = (originalExponent == (uint32_t)((1 << exponentBits) - 1));
86 
87  // Bit shift and edit until
88  uint32_t uSign = (value & signMask) << ((f32totBits - f32signBits) - (totalBits - signBits));
89  uint32_t uExponent = (originalExponent - exponentBias + f32expBias) << f32manBits;
90  uint32_t uMantissa = (value & mantMask) << (f32manBits - mantissaBits);
91 
92  // Make sure NaN/Infinity is preserved if original was NaN/Infinity.
93  if (originalIsNaN) { uExponent = ((1 << f32expBits) - 1) << f32manBits; }
94 
95  // Assign the expanded values to the union
96  intToFloat.uiVal = uSign | uExponent | uMantissa;
97 
98  // Return the expanded float
99  return intToFloat.fVal;
100 }

130 inline void convertXYZToCubeUV(float x, float y, float z, CubeFace& face, float& u, float& v)
131 {
132  float absX = fabs(x);
133  float absY = fabs(y);
134  float absZ = fabs(z);
135 
136  if (absX == 0.f && absY == 0.f && absZ == 0.f) { throw new std::runtime_error("Cannot convert the zero vector to a cubemap sample"); }
137 
138  int isXPositive = x > 0 ? 1 : 0;
139  int isYPositive = y > 0 ? 1 : 0;
140  int isZPositive = z > 0 ? 1 : 0;
141 
142  float maxAxis = 0.0f;
143  float uc = 0.0f;
144  float vc = 0.0f;
145 
146  // POSITIVE X
147  if (isXPositive && absX >= absY && absX >= absZ)
148  {
149  // u (0 to 1) goes from +z to -z
150  // v (0 to 1) goes from -y to +y
151  maxAxis = absX;
152  uc = -z;
153  vc = -y;
154  face = CubeFace::PositiveX;
155  }
156  // NEGATIVE X
157  if (!isXPositive && absX >= absY && absX >= absZ)
158  {
159  // u (0 to 1) goes from -z to +z
160  // v (0 to 1) goes from -y to +y
161  maxAxis = absX;
162  uc = z;
163  vc = -y;
164  face = CubeFace::NegativeX;
165  }
166  // POSITIVE Y
167  if (isYPositive && absY >= absX && absY >= absZ)
168  {
169  // u (0 to 1) goes from -x to +x
170  // v (0 to 1) goes from +z to -z
171  maxAxis = absY;
172  uc = x;
173  vc = z;
174  face = CubeFace::PositiveY;
175  }
176  // NEGATIVE Y
177  if (!isYPositive && absY >= absX && absY >= absZ)
178  {
179  // u (0 to 1) goes from -x to +x
180  // v (0 to 1) goes from -z to +z
181  maxAxis = absY;
182  uc = x;
183  vc = +z;
184  face = CubeFace::NegativeY;
185  }
186  // POSITIVE Z
187  if (isZPositive && absZ >= absX && absZ >= absY)
188  {
189  // u (0 to 1) goes from -x to +x
190  // v (0 to 1) goes from -y to +y
191  maxAxis = absZ;
192  uc = x;
193  vc = -y;
194  face = CubeFace::PositiveZ;
195  }
196  // NEGATIVE Z
197  if (!isZPositive && absZ >= absX && absZ >= absY)
198  {
199  // u (0 to 1) goes from +x to -x
200  // v (0 to 1) goes from -y to +y
201  maxAxis = absZ;
202  uc = -x;
203  vc = -y;
204  face = CubeFace::NegativeZ;
205  }
206 
207  // Convert range from -1 to 1 to 0 to 1
208  u = 0.5f * (uc / maxAxis + 1.0f);
209  v = 0.5f * (vc / maxAxis + 1.0f);
210 }
