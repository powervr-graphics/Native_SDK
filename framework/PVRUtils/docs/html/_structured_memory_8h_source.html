296  // INTERNAL NOTE: CALL THIS ** AFTER ** CALC BASE ALIGNMENT, TO FIX THE _offset AND _size MEMBERS
297  void calcSizeAndOffset(const uint32_t& offset)
298  {
299  _offset = align(offset, _baseAlignment);
300  _offset = align(_offset, _minDynamicAlignment);
301  if (isStructure())
302  {
303  uint32_t tmp_offset = 0;
304  for (auto& child : _childEntries)
305  {
306  child.calcSizeAndOffset(tmp_offset);
307  tmp_offset = child.getOffset() + static_cast<uint32_t>(child.getSize());
308  }
309  // STD140: Structures are padded to their alignment so that a[n] ==> sizeof(a[0]) == sizeof[a] / n
310  tmp_offset = align(tmp_offset, _baseAlignment);
311  tmp_offset = align(tmp_offset, _minDynamicAlignment);
312  _arrayMemberSize = tmp_offset;
313 
314  tmp_offset *= _numArrayElements;
315  _size = tmp_offset;
316  }
317  else
318  {
319  _arrayMemberSize = getSelfAlignedArraySize(_type);
320  _size = _variableArray ? _arrayMemberSize * _numArrayElements : pvr::getSize(_type, _numArrayElements);

624  StructuredBufferViewElement(const StructuredMemoryEntry& entry, uint32_t level, uint32_t elementArrayIndex, const uint32_t* parentIndices, uint32_t dynamicSlice = 0)
625  : _mappedMemory(nullptr), _level(level), _prototype(entry)
626  {
627  _indices[0] = elementArrayIndex;
628  if (parentIndices)
629  {
630  memcpy(_indices + 1, parentIndices, sizeof(uint32_t) * (level));
631  }
632  init(dynamicSlice);
633  }
634  void init(uint32_t dynamicSlice = 0)
635  {
636  // Get the offset INSIDE CURRENT LEVEL.
637  _offset = _prototype.getArrayElementOffset(_indices[0]);
638  const StructuredMemoryEntry* parent = _prototype.getParent();
639  size_t level = 1; // How many levels up we have gone
640  uint64_t dynamicSliceSize = 0;
641  uint32_t mappedDynamicSlice = 0;
642  while (parent) // Until we reach the root element
643  {
644  debug_assertion(parent->getNumArrayElements() > _indices[level], "StructuredBufferViewElement: Attempted out-of-bounds access in getOffset");
645  _offset += parent->getArrayElementOffset(_indices[level++]);
646  dynamicSliceSize = parent->getSize();
647  mappedDynamicSlice = parent->getMappedDynamicSlice();
648  // store the mapped memory so we only have to do this lookup once rather than each time setValue is called
649  _mappedMemory = parent->getMappedMemory();
650  parent = parent->getParent();
651  }
652 
653  // at this point dynamicSliceSize matches the root size
654  debug_assertion(dynamicSlice >= mappedDynamicSlice, "StructuredBufferViewElement: Mapped dynamic slice must be greater than or equal to the current dynamic slice");
655  uint32_t sliceOffset = (dynamicSlice * static_cast<uint32_t>(dynamicSliceSize)) - (mappedDynamicSlice * static_cast<uint32_t>(dynamicSliceSize));
656  _offset += sliceOffset;
657  }

938  {
939  if (this->_prototype.getPrimitiveType() != value.dataType() && value.dataType() != GpuDatatypes::mat3x3)
940  {
941  throw std::runtime_error("StructuredBufferView: Mismatched FreeValue datatype");
942  }
943  if (value.arrayElements() != _prototype.getNumArrayElements())
944  {
945  throw std::runtime_error("StructuredBufferView: Mismatched number of array elements");
946  }
947 
948  if (value.dataType() == GpuDatatypes::mat3x3)
949  {
950  size_t startoff = getOffset();
951  for (uint32_t i = 0; i < value.arrayElements(); ++i)
952  {
953  glm::mat3x4 tmp(value.interpretValueAs<glm::mat3x3>(i));
954 
955  uint64_t myoff = startoff + _prototype._arrayMemberSize * i;
956  uint64_t myvaluesize = getSize(value.dataType());
957  memcpy((char*)getMappedMemory() + (size_t)myoff, &tmp[0][0], (size_t)myvaluesize);
958  }
959  }
960  else
961  {
962  uint64_t myoff = getOffset();
963  uint64_t myvaluesize = value.dataSize();
964  memcpy((char*)getMappedMemory() + (size_t)myoff, value.raw(), (size_t)myvaluesize);
965  }
966  }

971  {

973  {
974  return _prototype.getNumChildren();
975  }

977  }
