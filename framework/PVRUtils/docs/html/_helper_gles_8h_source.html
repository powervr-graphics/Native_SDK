377  Area* insert(uint32_t width, uint32_t height)
378  {
379  // If this area has branches below it (i.e. is not a leaf) then traverse those.
380  // Check the left branch first.
381  if (left)
382  {
383  Area* tempPtr = NULL;
384  tempPtr = left->insert(width, height);
385  if (tempPtr != NULL) { return tempPtr; }
386  }
387  // Now check right
388  if (right) { return right->insert(width, height); }
389  // Already filled!
390  if (isFilled) { return NULL; }
391 
392  // Too small
393  if (size < width * height || w < width || h < height) { return NULL; }
394 
395  // Just right!
396  if (size == width * height && w == width && h == height)
397  {
398  isFilled = true;
399  return this;
400  }
401  // Too big. Split up.
402  if (size > width * height && w >= width && h >= height)
403  {
404  // Initializes the children, and sets the left child's coordinates as these don't change.
405  left = new Area;
406  right = new Area;
407  left->x = x;
408  left->y = y;
409 
410  // --- Splits the current area depending on the size and position of the placed texture.
411  // Splits vertically if larger free distance across the texture.
412  if ((w - width) > (h - height))
413  {
414  left->w = width;
415  left->h = h;
416 
417  right->x = x + width;
418  right->y = y;
419  right->w = w - width;
420  right->h = h;
421  }
422  // Splits horizontally if larger or equal free distance downwards.
423  else
424  {
425  left->w = w;
426  left->h = height;
427 
428  right->x = x;
429  right->y = y + height;
430  right->w = w;
431  right->h = h - height;
432  }
433 
434  // Initializes the child members' size attributes.
435  left->size = left->h * left->w;
436  right->size = right->h * right->w;
437 
438  // Inserts the texture into the left child member.
439  return left->insert(width, height);
440  }
441  // Catch all error return.
442  return NULL;
443  }

962 inline void create3dPlaneMesh(uint32_t width, uint32_t length, bool vertexAttribTex, bool vertexAttribNormal, assets::Mesh& outMesh)
963 {
964  const float halfWidth = width * .5f;
965  const float halfLength = length * .5f;
966 
967  glm::vec3 normal[4] = { glm::vec3(0.0f, 1.0f, 0.0f), glm::vec3(0.0f, 1.0f, 0.0f), glm::vec3(0.0f, 1.0f, 0.0f), glm::vec3(0.0f, 1.0f, 0.0f) };
968 
969  glm::vec2 texCoord[4] = {
970  glm::vec2(0.0f, 1.0f),
971  glm::vec2(0.0f, 0.0f),
972  glm::vec2(1.0f, 0.0f),
973  glm::vec2(1.0f, 1.0f),
974  };
975 
976  glm::vec3 pos[4] = { glm::vec3(-halfWidth, 0.0f, -halfLength), glm::vec3(-halfWidth, 0.0f, halfLength), glm::vec3(halfWidth, 0.0f, halfLength),
977  glm::vec3(halfWidth, 0.0f, -halfLength) };
978 
979  uint32_t indexData[] = { 0, 1, 2, 0, 2, 3 };
980 
981  float vertData[32];
982  uint32_t offset = 0;
983 
984  for (uint32_t i = 0; i < 4; ++i)
985  {
986  memcpy(&vertData[offset], &pos[i], sizeof(pos[i]));
987  offset += 3;
988  if (vertexAttribNormal)
989  {
990  memcpy(&vertData[offset], &normal[i], sizeof(normal[i]));
991  offset += 3;
992  }
993  if (vertexAttribTex)
994  {
995  memcpy(&vertData[offset], &texCoord[i], sizeof(texCoord[i]));
996  offset += 2;
997  }
998  }
999 
1000  uint32_t stride = sizeof(glm::vec3) + (vertexAttribNormal ? sizeof(glm::vec3) : 0) + (vertexAttribTex ? sizeof(glm::vec2) : 0);
1001 
1002  outMesh.addData((const uint8_t*)vertData, sizeof(vertData), stride, 0);
1003  outMesh.addFaces((const uint8_t*)indexData, sizeof(indexData), IndexType::IndexType32Bit);
1004  offset = 0;
1005  outMesh.addVertexAttribute("POSITION", DataType::Float32, 3, offset, 0);
1006  offset += sizeof(float) * 3;
1007  if (vertexAttribNormal)
1008  {
1009  outMesh.addVertexAttribute("NORMAL", DataType::Float32, 3, offset, 0);
1010  offset += sizeof(float) * 2;
1011  }
1012  if (vertexAttribTex) { outMesh.addVertexAttribute("UV0", DataType::Float32, 2, offset, 0); }
1013  outMesh.setPrimitiveType(PrimitiveTopology::TriangleList);
1014  outMesh.setStride(0, stride);
1015  outMesh.setNumFaces(ARRAY_SIZE(indexData) / 3);
1016  outMesh.setNumVertices(ARRAY_SIZE(pos));
1017 }
