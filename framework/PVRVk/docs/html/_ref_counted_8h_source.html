66  {

68  {

79  }

82  void decrement_count()
83  {
84  bool deleter = false; // DO NOT DESTROY THE OBJECT WHILE THE LOCK IS HELD!
85  int32_t cnt = 0;
86  {
87  if ((cnt = std::atomic_fetch_add(&_count, -1)) == 1)
88  {
89  std::lock_guard<std::recursive_mutex> lockguard(_mylock);
90  _is_deleting = true; // This flag tests notifies the decrement_weakcount that the object is deleting itself
91  destroyObject();

93  if (weakcount() == 0)
94  {
95  deleter = true;
96  }
97  }
98  }

101  deleteEntry();
