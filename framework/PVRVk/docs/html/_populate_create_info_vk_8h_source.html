92 inline void populateShaderInfo(const VkShaderModule& shaderModule, pvrvk::ShaderStageFlags shaderStageFlags, const std::string& entryPoint, const ShaderConstantInfo* shaderConsts,
93  uint32_t shaderConstCount, VkSpecializationInfo& specializationInfo, unsigned char* specializationInfoData, VkSpecializationMapEntry* mapEntries,
94  VkPipelineShaderStageCreateInfo& outShaderCreateInfo)
95 {
96  // caculate the number of size in bytes required.
97  uint32_t specicalizedataSize = 0;
98  for (uint32_t i = 0; i < shaderConstCount; ++i) { specicalizedataSize += shaderConsts[i].sizeInBytes; }
99 
100  if (specicalizedataSize)
101  {
102  assert(specicalizedataSize < FrameworkCaps::MaxSpecialisationInfoDataSize && "Specialised Data out of range.");
103  uint32_t dataOffset = 0;
104  for (uint32_t i = 0; i < shaderConstCount; ++i)
105  {
106  memcpy(&specializationInfoData[dataOffset], shaderConsts[i].data, shaderConsts[i].sizeInBytes);
107  mapEntries[i] = VkSpecializationMapEntry{ shaderConsts[i].constantId, dataOffset, shaderConsts[i].sizeInBytes };
108  dataOffset += shaderConsts[i].sizeInBytes;
109  }
110  specializationInfo.mapEntryCount = shaderConstCount;
111  specializationInfo.pMapEntries = mapEntries;
112  specializationInfo.dataSize = specicalizedataSize;
113  specializationInfo.pData = specializationInfoData;
114  }
115 
116  outShaderCreateInfo.sType = static_cast<VkStructureType>(pvrvk::StructureType::e_PIPELINE_SHADER_STAGE_CREATE_INFO);
117  outShaderCreateInfo.pNext = nullptr;
118  outShaderCreateInfo.flags = static_cast<VkPipelineShaderStageCreateFlags>(pvrvk::PipelineShaderStageCreateFlags::e_NONE);
119  outShaderCreateInfo.pSpecializationInfo = (specicalizedataSize ? &specializationInfo : nullptr);
120  outShaderCreateInfo.stage = static_cast<VkShaderStageFlagBits>(shaderStageFlags);
121  outShaderCreateInfo.module = shaderModule;
122  outShaderCreateInfo.pName = entryPoint.c_str();
123 }
