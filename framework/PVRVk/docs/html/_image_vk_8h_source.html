13 inline ImageViewType convertToPVRVkImageViewType(ImageType baseType, uint32_t numArrayLayers, bool isCubeMap)
14 {
15  // if it is a cube map it has to be 2D Texture base
16  if (isCubeMap && baseType != ImageType::e_2D)
17  {
18  assert(baseType == ImageType::e_2D && "Cubemap texture must be 2D");
19  return ImageViewType::e_MAX_ENUM;
20  }
21  // array must be atleast 1
22  if (!numArrayLayers)
23  {
24  assert(false && "Number of array layers must be greater than equal to 0");
25  return ImageViewType::e_MAX_ENUM;
26  }
27  // if it is array it must be 1D or 2D texture base
28  if ((numArrayLayers > 1) && (baseType > ImageType::e_2D))
29  {
30  assert(false && "1D and 2D image type supports array texture");
31  return ImageViewType::e_MAX_ENUM;
32  }
33 
34  ImageViewType vkType[] = { ImageViewType::e_1D, ImageViewType::e_1D_ARRAY, ImageViewType::e_2D, ImageViewType::e_2D_ARRAY, ImageViewType::e_3D, ImageViewType::e_CUBE,
35  ImageViewType::e_CUBE_ARRAY };
36  if (isCubeMap) { numArrayLayers = (numArrayLayers > 6u) * 6u; }
37  return vkType[(static_cast<uint32_t>(baseType) * 2) + (isCubeMap ? 3 : 0) + (numArrayLayers > 1 ? 1 : 0)];
38 }

40 inline ImageAspectFlags formatToImageAspect(Format format)
41 {
42  if (format == Format::e_UNDEFINED) { throw ErrorUnknown("Cannot retrieve VkImageAspectFlags from an undefined VkFormat"); }
43  if (format < Format::e_D16_UNORM || format > Format::e_D32_SFLOAT_S8_UINT) { return ImageAspectFlags::e_COLOR_BIT; }
44  const ImageAspectFlags formats[] = {
45  ImageAspectFlags::e_DEPTH_BIT, // VkFormat::e_D16_UNORM
46  ImageAspectFlags::e_DEPTH_BIT, // VkFormat::e_X8_D24_UNORM_PACK32
47  ImageAspectFlags::e_DEPTH_BIT, // VkFormat::e_D32_SFLOAT
48  ImageAspectFlags::e_STENCIL_BIT, // VkFormat::e_S8_UINT
49  ImageAspectFlags::e_DEPTH_BIT | ImageAspectFlags::e_STENCIL_BIT, // VkFormat::e_D16_UNORM_S8_UINT
50  ImageAspectFlags::e_DEPTH_BIT | ImageAspectFlags::e_STENCIL_BIT, // VkFormat::e_D24_UNORM_S8_UINT
51  ImageAspectFlags::e_DEPTH_BIT | ImageAspectFlags::e_STENCIL_BIT, // VkFormat::e_D32_SFLOAT_S8_UINT
52  };
53  return formats[static_cast<uint32_t>(format) - static_cast<uint32_t>(Format::e_D16_UNORM)];
54 }
