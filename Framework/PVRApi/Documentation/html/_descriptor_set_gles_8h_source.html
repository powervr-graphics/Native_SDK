62 {
63  platform::ContextGles& contextEs = static_cast<platform::ContextGles&>(context);
64 
65  const TextureStore& resource = view->getResource();
66 
67  if (resource.isNull())
68  {
69  Log("TextureView_::bind attempted to bind a texture with NULL native texture handle");

71  }
72  platform::ContextGles::RenderStatesTracker& renderStates = contextEs.getCurrentRenderStates();
73
