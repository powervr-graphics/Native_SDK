707 /*!*********************************************************************************************************************
708 \brief The Sync object can be used both as an API command, or to be directly queried by the application.
709 Each time the CommandBuffer is submitted, an underlying sync object is added to the list. If the MaxSize is reached, the
710 first sync object to be submitted is discarder. The user normally queries/uses the syncs last-to-first with an index. This
711 technique makes it trivial to use frame-lag techniques, like double/triple buffering.
712 Each Sync starts in a non-signaled state, and becomes Signaled as soon as the GPU actually executes it previous command.
713 All of the functions work on a last-to-first indexing - passing 0 refers to the last command buffer submission, and from
714 there backwards in time, until getMaxSize() - 1, which is the earliest sync point used.
715 Use:
716 Get a sync object with insertFenceSync().
717 2) Use clientWait() to CPU-block until the Sync becomes Signaled.
718 3) Use isSignaled() to query if the Sync is Signaled, without blocking.
719 4) Use commandBuffer.serverSync() to force the implementation to wait for the specified sync object before proceeding.
720 ***********************************************************************************************************************/
721 class Sync_
722 {
723  friend class ::pvr::api::impl::CreateFenceSync_;
724  friend class ::pvr::api::impl::CommandBufferBase_;
725  template<typename> friend struct ::pvr::RefCountEntryIntrusive;
726  pvr::RingBuffer<void*> pimpl;
727  uint32 maxSize;
728  Sync_();
729  ~Sync_();
730  void serverWait(uint32 which);
731 public:
732 
733  /*!*********************************************************************************************************************
734  \brief Manually discards the last howMany sync points. Specialised use only.
735  \param howMany The number of sync points to discard. Non positive numbers have no effect.
736  ***********************************************************************************************************************/
737  void discardLast(int32 howMany = 1);
738 
739  /*!*********************************************************************************************************************
740  \brief Get the maximum number of syncs that will be created. Each time the command buffer that created this sync object
741  (with the insertFenceSync call) is submitted, another sync is created. When MaxSize is reached, the first sync to be
742  created is discarded.
743  \return The maximum number of sync points that this object represents.
744  ***********************************************************************************************************************/
745  uint32 getMaxSize() { return this->maxSize; }
746 
747  /*!*********************************************************************************************************************
748  \brief Set the maximum number of syncs that will be created. Each time the command buffer that created this sync object
749  (with the insertFenceSync call) is submitted, another sync is created. When MaxSize is reached, the first sync to be
750  created is discarded.
751  \param maxSize The maximum number of sync points that this object represents will be set to this number
752  ***********************************************************************************************************************/
753  void setMaxSize(uint32 maxSize) { this->maxSize = maxSize; }
754 
755  /*!*********************************************************************************************************************
756  \brief Call this function to test if a Sync is signalled, without blocking for it. Each call to "submit" on the command
757  buffer that created this sync, will push a new sync point to the front of this queue.
758  \param which The ordinal, newest-to-oldest, of the sync point to query (0=last submit, 1=previous frame etc.)
759  \return True if the Sync is signaled (hence its previous commands are complete), false otherwise
760  ***********************************************************************************************************************/
761  bool isSignaled(uint32 which);
762 
763  /*!*********************************************************************************************************************
764  \brief Call this function to wait on the sync object (i.e. wait for it to become Signalled, meaning that any commands
765  preceding it are complete)
766  \param timeout Optional: Provide a timeout. If 0 is passed, the wait is infinite.
767  \param which The fence sync to wait on (0 is the latest, maxSize is the earliest).
768  \return * SyncResult::Ok if the Sync was either already signaled or became signaled within the timeout (the preceding
769  commands have finished)
770  * SyncResult::CommandBufferNotSubmitted if the commandBuffer that would generat the Sync has not yet been
771  submitted, hence the sync will never be signalled (so waiting on it would probably result in a deadlock).
772  * SyncResult::TimeoutExpired, if the Sync was NOT signalled, but the timeout expired.
773  * SyncResult::Failed, if the implementation failed to wait (for example, the sync was already destroyed)
774  ***********************************************************************************************************************/
775  SyncWaitResult::Enum clientWait(uint32 which, uint64 timeout = 0);
776 };
