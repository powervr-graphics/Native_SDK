268  void enableAttribute(uint16 attributeIdx)
269  {
270  m_renderStatesTracker.attributesToEnableBitfield |= (1 << attributeIdx);
271  if ((m_renderStatesTracker.attributesEnabledBitfield & (1 << attributeIdx)) == 0)
272  {
273  gl::EnableVertexAttribArray(attributeIdx);
274  m_renderStatesTracker.attributesEnabledBitfield |= (1 << attributeIdx);
275  }
276  m_renderStatesTracker.attributesMaxToEnable = (std::max)(attributeIdx, m_renderStatesTracker.attributesMaxToEnable) + 1;
277  }
278 
279  /*!*********************************************************************************************************************
280  \brief Internal use. State tracking.disable unneeded vertex attributes.
281  ***********************************************************************************************************************/
282  void disableUnneededAttributes()
283  {
284  uint32 deactivate = m_renderStatesTracker.attributesEnabledBitfield ^ m_renderStatesTracker.attributesToEnableBitfield;
285 
286  for (uint8 i = 0; i < m_renderStatesTracker.attributesMaxEnabled; ++i)
287  {
288  if (Bitfield<uint32>::isSet(deactivate, i))
289  {
290  gl::DisableVertexAttribArray(i);
291  }
292  }
293  m_renderStatesTracker.attributesEnabledBitfield = m_renderStatesTracker.attributesToEnableBitfield;
294  m_renderStatesTracker.attributesMaxEnabled = m_renderStatesTracker.attributesMaxToEnable;
295  m_renderStatesTracker.attributesMaxToEnable = 0;
296  m_renderStatesTracker.attributesToEnableBitfield = 0;
297  }
