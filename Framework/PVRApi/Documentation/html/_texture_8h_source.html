420 /*!****************************************************************************************************************
421 \brief Get the display's color, depth and or stencil format from a DisplayAttributes object.
422 \param[in] attribute The DisplayAttributes to query. Normally from shell.getDisplayAttributes()
423 \param[out] outColorFmt The Color Format that this DisplayAttributes object has
424 \param[out] outDepthStencilFmt The DepthStencilFormat that this DisplayAttributes object has
425 \description This function is usually called on an object found with Shell::getDisplayAttributes(), in order to
426 determine the formats that are necessary for an On Screen FBO.
427 *******************************************************************************************************************/
428 inline void getDisplayFormat(const system::DisplayAttributes& attribute,
429  api::ImageDataFormat* outColorFmt,
430  api::ImageDataFormat* outDepthStencilFmt)
431 {
432  if (outColorFmt)
433  {
434  outColorFmt->format = PixelFormat(attribute.redBits ? 'r' : 0, attribute.greenBits ? 'g' : 0,
435  attribute.blueBits ? 'b' : 0, attribute.alphaBits ? 'a' : 0,
436  (uint8)attribute.redBits, (uint8)attribute.greenBits,
437  (uint8)attribute.blueBits, (uint8)attribute.alphaBits);
438  }
439  if (outDepthStencilFmt)
440  {
441  outDepthStencilFmt->format = PixelFormat('d', (attribute.stencilBPP ? 's' : 0), (uint8)0, (uint8)0,
442  (uint8)attribute.depthBPP, (uint8)attribute.stencilBPP, (uint8)0, (uint8)0);
443  }
444 }
