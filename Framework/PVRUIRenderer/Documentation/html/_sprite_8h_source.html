724 /*!****************************************************************************************************************
725 \brief Abstract container for sprites. See MatrixGroup or PixelGroup. A group contains references to a number
726  of sprites, allowing hierarchical transformations to be applied to them
727 \details A very complex transformation using the "id" member is used to optimize the group transformations, as
728  each child needs to hold the transformations
729 *******************************************************************************************************************/
730 class Group_ : public Sprite_
731 {
732 protected:
734  struct SpriteEntryEquals
735  {
736  Sprite sprite;
737  SpriteEntryEquals(const Sprite& sprite) : sprite(sprite) {}
738  bool operator()(const Sprite& rhs) { return sprite == rhs; }
739  };
740  typedef std::vector<Sprite> ChildContainer;
741  mutable ChildContainer m_children;
742  pvr::uint64 m_id;
744 public:
745  /*!****************************************************************************************************************
746  \brief Constructor. Internal use. The parameter groupid is an implementation detail used to implement and optimize
747  group behaviour, and cannot be trivially determined.
748  *******************************************************************************************************************/
749  Group_(UIRenderer& uiRenderer, pvr::uint64 groupid) : Sprite_(uiRenderer), m_id(groupid) { }
750 
751  /*!****************************************************************************************************************
752  \brief Add a Sprite (Text, Image etc.) to this Group. All sprites in the group will be transformed together when
753  calling Render on the group.
754  \param sprite The Sprite to add.
755  \return Pointer to this object, in order to easily chan add commands.
756  *******************************************************************************************************************/
757  Group_* add(const Sprite& sprite)
758  {
759  m_children.push_back(sprite);
760  m_boundingRect.add(sprite->getDimensions().x, sprite->getDimensions().y, 0.0f);
761  return this;
762  }
763 
764  /*!****************************************************************************************************************
765  \brief Remove a Sprite from this Group. Linear search (Complexity O(n) )
766  \param sprite The Sprite to remove.
767  *******************************************************************************************************************/
768  void remove(const Sprite& sprite)
769  {
770  ChildContainer::iterator it = std::find_if(m_children.begin(), m_children.end(), SpriteEntryEquals(sprite));
771  if (it != m_children.end()) {
772  m_boundingRect.remove((*it)->getBoundingBox());
773  m_children.erase(it);
774  }
775  }
776 
777  /*!****************************************************************************************************************
778  \internal
779  \brief Internal function. Do not call directly. Triggered when all item transformations have been applied and the
780  final matrices can be calculated.
781  *******************************************************************************************************************/
782  void calculateMvp(pvr::uint64 parentIds, glm::mat4 const& srt, const glm::mat4& viewProj,
783  pvr::Rectanglei const& viewport)const
784  {
785  glm::mat4 tmpMatrix = srt * m_cachedMatrix;
786  //My cached matrix should always be up-to-date unless overridden. No effect.
787  for (ChildContainer::iterator it = m_children.begin(); it != m_children.end(); ++it)
788  {
789  (*it)->calculateMvp(packId(parentIds, m_id), tmpMatrix, viewProj, viewport);
790  }
791  }
792 
793 
794  /*!****************************************************************************************************************
795  \internal
796  \brief Internal function that UIRenderer calls to render. Do not call directly.
797  *******************************************************************************************************************/
798  virtual void onRender(api::SecondaryCommandBuffer& commandBuffer, pvr::uint64 parentId) const
799  {
800  for (ChildContainer::iterator it = m_children.begin(); it != m_children.end(); ++it)
801  {
802  (*it)->onRender(commandBuffer, packId(parentId, m_id));
803  }
804  }
805 protected:
806  pvr::uint64 packId(pvr::uint64 parentIds, pvr::uint64 id)const
807  {
808  pvr::uint64 packed = parentIds << NUM_BITS_GROUP_ID;
809  return packed |= id;
810  }
811 };
