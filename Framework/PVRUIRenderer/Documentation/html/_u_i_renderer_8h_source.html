112  Result::Enum init(GraphicsContext& context, const api::RenderPass& renderpass,
113  uint32 subpass)
114  {
115  release();
116  m_mustEndCommandBuffer = false;
117  m_context = context;
118  m_screenDimensions = glm::vec2(context->getDisplayAttributes().width, context->getDisplayAttributes().height);
119  m_renderpass = renderpass;
120  m_subpass = subpass;
121  // screen rotated?
122  if (m_screenDimensions.y > m_screenDimensions.x && context->getDisplayAttributes().fullscreen)
123  {
124  rotateScreen90degreeCCW();
125  }
126 
127 
128  Result::Enum res;
129  if (((res = init_CreateDescriptorSetLayout()) == Result::Success) &&
130  ((res = init_CreatePipelineAndRenderPass()) == Result::Success))
131  {
132  if (!init_CreateDefaultSampler()) { res = Result::UnknownError; }
133  if (!init_CreateDefaultSdkLogo()) { res = Result::UnknownError; }
134  if (!init_CreateDefaultFont()) { res = Result::UnknownError; }
135  if (!init_CreateDefaultTitle()) { res = Result::UnknownError; }
136  }
137  return res;
138  }

293  /*!****************************************************************************************************************
294  \brief Begin rendering to a specific CommandBuffer. Must be called to render sprites. DO NOT update sprites after
295  calling this function before calling endRendering.
296  \param cb The pvr::api::SecondaryCommandBuffer object where all the rendering commands will be put into.
297  \description THIS METHOD OR ITS OVERLOAD MUST BE CALLED BEFORE RENDERING ANY SPRITES THAT BELONG TO A SPECIFIC
298  UIRenderer. The sequence must always be beginRendering, render ..., endRendering. Always try to group
299  as many rendering commands as possible between begin and end, to avoid needless state changes.
300  *******************************************************************************************************************/
301  void beginRendering(api::SecondaryCommandBuffer cb)
302  {
303  if (!cb->isRecording())
304  {
305  cb->beginRecording(m_renderpass, m_subpass);
306  m_mustEndCommandBuffer = true;
307  }
308  else { m_mustEndCommandBuffer = false; }
309  cb->pushPipeline();// store the currently bound pipeline
310  cb->bindPipeline(getPipeline());// bind the uirenderer pipeline
311  m_activeCommandBuffer = cb;
312  }
313

316  \param cb The pvr::api::CommandBuffer object where all the rendering commands will be put into.
317  \param pipe The pvr::api::GraphicsPipeline to use for rendering.
318  \description THIS METHOD OR ITS OVERLOAD MUST BE CALLED BEFORE RENDERING ANY SPRITES THAT BELONG TO A SPECIFIC
319  UIRenderer. The sequence must always be beginRendering, render ..., endRendering. Always try to group
320  as many rendering commands as possible between beginRendering and endRendering, to avoid needless
321  state changes. Use this overload to render with a custom GraphicsPipeline.
322  *******************************************************************************************************************/
323  void beginRendering(api::SecondaryCommandBuffer cb, pvr::api::GraphicsPipeline& pipe)
324  {
325  if (!cb->isRecording())
326  {
327  cb->beginRecording(m_renderpass, m_subpass);
328  m_mustEndCommandBuffer = true;
329  }
330  else { m_mustEndCommandBuffer = false; }
331  cb->pushPipeline();
332  cb->bindPipeline(pipe);
333  m_activeCommandBuffer = cb;

336  /*!****************************************************************************************************************
337  \brief End rendering. Always call this method before submitting the commandBuffer passed to the UIRenderer.
338  \description This method must be called after you finish rendering sprites (after a call to beginRendering). The
339  sequence must always be beginRendering, render ..., endRendering. ry to group as many of the rendering
340  commands (preferably all) between beginRendering and endRendering.
341  *******************************************************************************************************************/
342  void endRendering()
343  {
344  if (m_activeCommandBuffer.isValid())
345  {
346  m_activeCommandBuffer->popPipeline();
347  if (m_mustEndCommandBuffer)
348  {
349  m_mustEndCommandBuffer = false;
350  m_activeCommandBuffer->endRecording();
351  }
352  m_activeCommandBuffer.release();
353  }
354  }
