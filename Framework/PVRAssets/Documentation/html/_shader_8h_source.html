164  /*!*******************************************************************************************************************************
165  \brief Populate a list of valid shader version for a given file
166  \return Return number of version populated.
167  \param filename shader file name to populate
168  \param assetProvider asset loader
169  **********************************************************************************************************************************/
170  int populateValidVersions(const std::string& filename, IAssetProvider& assetProvider)
171  {
172  m_filenames.clear();
173  m_assetProvider = &assetProvider;
174 
175  //Case 1: The filename already refers to a specific API. In this case, as soon as it
176  //is found, stop. We won't bother with others.
177  std::string name, extension;
178  strings::getFileNameAndExtension(filename, name, extension);
179  for (int i = 1; i < Api::Count; ++i)
180  {
181  if (strings::endsWith(name, Api::getApiCode(Api::Enum(i))))
182  {
183  Stream::ptr_type str = m_assetProvider->getAssetStream(filename, false);
184  if (str.get())
185  {
186  setFileForApi(filename, Api::Enum(i));
187  return 1; //This file has a suffix, so it only has 1 api
188  }
189  return 0; //Filename has a suffix but was not found... Let's not go over the top here, assume this is failure.
190  }
191  }
192  int count = 0;
193 
194  //Being here means that the filename did not have an API suffix. Which is the main case: Load everything you find.
195  //So now do the main work: Test all possible Apis...
196  for (int i = 1; i < Api::Count; ++i)
197  {
198  string file1 = name + "_" + Api::getApiCode(Api::Enum(i)) + (extension.size() ? "." + extension : "");
199 
200  Stream::ptr_type str = m_assetProvider->getAssetStream(file1, false);
201  if (str.get())
202  {
203  setFileForApi(file1, Api::Enum(i));
204  ++count;
205  continue;
206  }
207  file1 = name + Api::getApiCode(Api::Enum(i)) + (extension.size() ? "." + extension : "");
208  str = m_assetProvider->getAssetStream(file1, false);
209  if (str.get())
210  {
211  setFileForApi(file1, Api::Enum(i));
212  ++count;
213  }
214  }
215 
216  //Lastly: If we still cannot find a file, there is a last case: Legacy (the user is using a file without extension).
217  //In this case, we will load it as minimum version (ES2) and get it over with.
218  if (count == 0)
219  {
220  Stream::ptr_type str = m_assetProvider->getAssetStream(filename, false);
221  if (str.get())
222  {
223  setFileForApi(filename, Api::OpenGLES2);
224  return 1; //This file has a suffix, so it only has 1 api
225  }
226  }
227  if (count == 0)
228  {
229  Log(Log.Error, "ShaderFile::populateValidVersions: No valid files found for filename [%s]", filename.c_str());
230  }
231  return count;
232  }
233 };
