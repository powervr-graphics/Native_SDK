143  int populateValidVersions(const std::string& filename, IAssetProvider& assetProvider)
144  {
145  _filenames.clear();
146  _assetProvider = &assetProvider;
147 
148  //Case 1: The filename already refers to a specific API. In this case, as soon as it
149  //is found, stop. We won't bother with others.
150  std::string name, extension;
151  strings::getFileNameAndExtension(filename, name, extension);
152  for (int i = 1; i < (int)Api::NumApis; ++i)
153  {
154  if (strings::endsWith(name, apiCode(Api(i))))
155  {
156  Stream::ptr_type str = _assetProvider->getAssetStream(filename, false);
157  if (str.get())
158  {
159  setFileForApi(filename, Api(i));
160  return 1; //This file has a suffix, so it only has 1 api
161  }
162  return 0; //Filename has a suffix but was not found... Let's not go over the top here, assume this is failure.
163  }
164  }
165  int count = 0;
166 
167  //Being here means that the filename did not have an API suffix. Which is the main case: Load everything you find.
168  //So now do the main work: Test all possible Apis...
169  for (int i = 1; i < (int)Api::NumApis; ++i)
170  {
171  std::string file1 = name + "_" + apiCode(Api(i)) + (extension.size() ? "." + extension : "");
172  if (i == (int)Api::Vulkan) { file1 += ".spv"; }
173  Stream::ptr_type str = _assetProvider->getAssetStream(file1, false);
174  if (str.get())
175  {
176  setFileForApi(file1, Api(i));
177  ++count;
178  continue;
179  }
180  file1 = name + apiCode(Api(i)) + (extension.size() ? "." + extension : "");
181  str = _assetProvider->getAssetStream(file1, false);
182  if (str.get())
183  {
184  setFileForApi(file1, Api(i));
185  ++count;
186  }
187  }
188 
189  //Lastly: If we still cannot find a file, there is a last case: Legacy (the user is using a file without extension).
190  //In this case, we will load it as minimum version (ES2) and get it over with.
191  if (count == 0)
192  {
193  Stream::ptr_type str = _assetProvider->getAssetStream(filename, false);
194  if (str.get())
195  {
196  setFileForApi(filename, Api::OpenGLES2);

198  }
