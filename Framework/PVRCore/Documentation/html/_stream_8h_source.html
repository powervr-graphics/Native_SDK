111  /*!********************************************************************************************************
112  \brief Convenience functions that reads all data in the stream into a contiguous block of memory of
113  a specified element type.
114  Requires random-access stream.
115  \tparam Type_ The type of item that will be read into.
116  \return A std::vector<Type_> containing all data from the current point to the end of the stream.
117  **********************************************************************************************************/
118  template<typename Type_> std::vector<Type_> readToEnd()
119  {
120  std::vector<Type_> ret;
121  size_t mySize = getSize();
122  size_t myElements = mySize / sizeof(Type_);
123  ret.resize(mySize);
124  size_t actuallyRead;
125  read(sizeof(Type_), myElements, ret.data(), actuallyRead);
126  return ret;
127  }

129  /*!********************************************************************************************************
130  \brief Convenience function that reads all data in the stream into a raw, contiguous block of memory.
131  Requires random-access stream.
132  \param[out] outString A std::vector<char> that will contain all data in the stream.
133  \return pvr::Result::Success if successful, error code if otherwise.
134  **********************************************************************************************************/
135  bool readIntoCharBuffer(std::vector<char>& outString) const
136  {
137  bool rslt = Result::Success;
138  if (!isopen()) { return false; }
139  PVR_ASSERT(getSize() < (std::numeric_limits<size_t>::max)());
140  outString.resize(getSize() + 1);
141 
142  size_t dataRead;
143  rslt = read(1, getSize(), outString.data(), dataRead);
144  if (rslt != Result::Success)
145  {
146  return rslt;
147  }
148 
149  outString[dataRead] = '\0';
150  return rslt;
151  }
