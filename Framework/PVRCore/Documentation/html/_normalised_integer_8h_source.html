325 template <typename IntegerType>
326 void NormalisedInteger<IntegerType>::packDouble(double value)
327 {
328  // Check if the value needs to be clamped.
329  if (value > 1.0)
330  {
331  m_value = std::numeric_limits<IntegerType>::max();
332  }
333  else if (value <= -1.0)
334  {
335  m_value = std::numeric_limits<IntegerType>::min();
336  }
337  else
338  {
339  m_value = static_cast<IntegerType>(value * static_cast<double>(std::numeric_limits<IntegerType>::max()));
340  }
341 }

343 template <typename IntegerType>
344 double NormalisedInteger<IntegerType>::unpackDouble() const
345 {
346  double value = (static_cast<double>(m_value) / static_cast<double>(std::numeric_limits<IntegerType>::max()));
347 
348  if (value > 1.0)
349  {
350  return 1.0;
351  }
352  else if (value < -1.0)
353  {
354  return -1.0;
355  }
356  else
357  {
358  return value;
359  }
