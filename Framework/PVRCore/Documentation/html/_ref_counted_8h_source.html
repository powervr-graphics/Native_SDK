41  void increment_count()
42  {
43  if (std::atomic_fetch_add(&count_, 1) == 0)
44  {
45  std::lock_guard<std::mutex> lockguard(mylock);
46  if (count_.load() == 0)

49  }
50  }

54  {
55  if (std::atomic_fetch_add(&weakcount_, 1) == 0)
56  {
57  std::lock_guard<std::mutex> lockguard(mylock);

61  }

68  {
69  bool deleter = false; //Track, if it was THIS object that was the one to do the deletion!
