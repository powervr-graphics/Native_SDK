418  void compact()
419  {
420  //We can do that because the last remove() tears down all datastructures used.
421  if (!myindex.size())
422  {
423  return;
424  }
425  //First, make sure there is something to compact...
426  deleteditemlisttype_::iterator unused_spot = myDeletedItems.begin();
427  while (myDeletedItems.size() && unused_spot != myDeletedItems.end())
428  {
429  //Last item in the storage vector.
430  size_t last = mystorage.size();
431 
432  //1) Trim the end of the vector...
433  while (last-- && mystorage[last].isUnused)
434  {
435  mystorage.pop_back();
436  //Rinse, repeat. If size is zero, there is nothing to do.
437  }
438 
439  //Either the storage is empty
440  if (mystorage.empty())
441  {
442  myDeletedItems.clear();
443  //the rest should already be cleared...
444  //the loop will exit naturally
445  }
446  else //Or we can have find its last item!
447  {
448  //2)Trim any items from the end of the unused spots list that may have been trimmed off by the vector...
449  while (unused_spot != myDeletedItems.end() && *unused_spot >= last)
450  {
451  myDeletedItems.erase(unused_spot++);
452  }
453  //Any spots left?
454  if (unused_spot != myDeletedItems.end())
455  {
456  //Do the actual data movement. After all we've been through, we know that
457  //i. The last item of the vector is a valid item (guaranteed by 1)
458  //ii. The unused spot is not out of bounds of the vector
459  //iii.
460  //Copy by hand(as we have not defined a move assignment operator for compatibility reasons.
461  //Also, since the std::string does not throw, this makes easier to reason about exceptions.
462 
463  mystorage[*unused_spot].value = std::move(mystorage[last].value);
464  mystorage[*unused_spot].key = std::move(mystorage[last].key);
465 
466  mystorage[*unused_spot].isUnused = false;
467  myindex[mystorage[*unused_spot].key] = *unused_spot;
468  mystorage.pop_back();
469  myDeletedItems.erase(unused_spot++);
470  }//else : No action needed - unused spots has been trimmed off completely, so no movement is possible, or necessary...

516  bool relocate(const IndexType_& key, size_t index)
517  {
518  typename maptype_::iterator found = myindex.find(key);
519  if (found == myindex.end()) { return false; }
520  size_t old_index = myindex[key];
521  if (index == old_index) { return true; } //No-op
522  if (index + 1 > mystorage.size()) // Storage not big enough.
523  {
524  //Grow, and mark unused all required items. Need to add all spots (But the last) to unusedspots.
525  size_t oldsize = mystorage.size();
526  mystorage.resize(index + 1);
527  for (size_t i = oldsize; i < index; ++i)
528  {
529  myDeletedItems.push_front(i);
530  mystorage[i].isUnused = 1;
531  }
532  mystorage.back() = mystorage[old_index];
533  removefromvector(old_index);
534  }
535  else if (mystorage[index].isUnused) // Lucky! Storage is big enough, and the item is not used!
536  {
537  deleteditemlisttype_::iterator place = std::find(myDeletedItems.begin(), myDeletedItems.end(), index);
538  assert(place != myDeletedItems.end()); // Shouldn't happen! Ever!
539  myDeletedItems.erase(place);
540  mystorage[index] = mystorage[old_index];
541  removefromvector(old_index);
542  }
543  else // Whoops! Space is already occupied. Swap with the old item!
544  {
545  myindex[mystorage[index].key] = old_index;
546  std::swap(mystorage[index], mystorage[old_index]);
547  }
548  myindex[key] = index;
549  return true;
550  }
