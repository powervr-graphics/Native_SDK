502 /**********************************************************************************************
503 \brief Aligns an offset (previousTotalSize) with the alignment of a type. It's called
504 getOffsetAfter, because it is commonly used to find the offset of an item when the previous
505 total size is known (i.e. align a vec4 to a struct whose previous size was 30 bytes? returns
506 32 bytes...)
507 **********************************************************************************************/
508 inline uint32 getOffsetAfter(GpuDatatypes::Enum type, uint32 previousTotalSize)
509 {
510  uint32 align = getAlignment(type);
511 
512  uint32 diff = previousTotalSize % align;
513  diff += (align * (diff == 0)); // REMOVE BRANCHING -- equal to : if(diff==0) { diff+=8 }
514  return previousTotalSize - diff + align;
515 }

517 /**********************************************************************************************
518 \brief Returns the new size of a hypothetical struct whose old size was previousTotalSize,
519 and to which "arrayElement" new items of type "type" are added
520 **********************************************************************************************/
521 inline uint32 getTotalSizeAfter(GpuDatatypes::Enum type, uint32 arrayElements, uint32 previousTotalSize)
522 {
523  PVR_ASSERTION(arrayElements > 0);
524  // Arrays pads their last element to their alignments. Standalone objects do not. vec3[3] is NOT the same as vec3;vec3;vec3;
525  uint32 selfAlignedSize = getSelfAlignedArraySize(type) * arrayElements * (arrayElements != 1); //Equivalent to: if (arrayElements!=1) selfAlign = getSelfAl... else selfAlign=0
526  // Other elements do not.
527  uint32 unalignedSize = getSize(type) * (arrayElements == 1);//Equivalent to: if (arrayElements==1) unaligned = getUnaligned.... else unaligned=0
528 
529  return getOffsetAfter(type, previousTotalSize) + selfAlignedSize + unalignedSize;
530 }
